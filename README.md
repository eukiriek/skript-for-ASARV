# 0. На всякий случай приводим нужные поля к timedelta
for col in ["Отработанное время",
            "Общая продолжительность",
            "Время отсутствия",
            "Время обеда"]:
    df[col] = pd.to_timedelta(df[col], errors="coerce")

# 1. Если Отработанное время >= 24 часов
mask_ge_24 = df["Отработанное время"] >= pd.Timedelta(hours=24)

# считаем максимум из "Время отсутствия" и "Время обеда"
max_break_24 = df[["Время отсутствия", "Время обеда"]].max(axis=1)

# NaT в перерывах считаем как 0
max_break_24 = max_break_24.fillna(pd.Timedelta(0))

# Новое отработанное время = Общая продолжительность - max(отсутствие, обед)
df.loc[mask_ge_24, "Новое отработанное время"] = (
    df.loc[mask_ge_24, "Общая продолжительность"] - max_break_24[mask_ge_24]
)
