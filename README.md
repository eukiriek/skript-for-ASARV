# Приводим колонку к строкам (нужно для корректной проверки)
col = "Время отсутствия"
df[col] = df[col].astype(str).str.strip()

# Маска пустых, нулевых или невалидных значений
mask_empty = (
    df[col].isna() |
    (df[col] == "") |
    (df[col] == "0") |
    (df[col] == "0:00") |
    (df[col] == "00:00") |
    (df[col] == "0:00:00") |
    (df[col] == "00:00:00")
)

# Заполняем только пустые — остальные НЕ трогаем
df.loc[mask_empty, col] = "00:00:00"
